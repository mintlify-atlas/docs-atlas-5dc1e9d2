# Desktop Mode

Desktop Mode provides a macOS-like graphical interface with draggable desktop icons, interactive windows, and a floating dock for quick navigation.

## Overview

When you load the portfolio, Desktop Mode is the default view. It features:
- Desktop icons arranged in columns
- Floating dock at the bottom
- Interactive windows
- Customizable wallpapers
- Smooth animations and transitions

## Desktop Icons

### Icon Interaction

Icons support both single-click (mobile/tablet) and double-click (desktop) interactions:

```tsx
// From DesktopIcons.tsx:200-237
const handleIconClick = (
  e: React.MouseEvent | React.KeyboardEvent,
  iconName: string,
  action?: () => void
) => {
  const isKeyboard = (e as React.MouseEvent).detail === 0;
  
  // Mobile/tablet: Open immediately
  if (deviceType === 'mobile' || deviceType === 'tablet' || isKeyboard) {
    openApp(iconName, action);
    return;
  }
  
  // Desktop: Double-click required
  if (selectedIcon === iconName) {
    openApp(iconName, action);
    setSelectedIcon(null);
  } else {
    // Show "Double-click to open" tooltip
    setSelectedIcon(iconName);
    setClickHelpIcon(iconName);
    // Auto-clear after 650ms
    setTimeout(() => {
      setSelectedIcon(null);
    }, 650);
  }
};
```

## Responsive Behavior

### Device Detection

```tsx
// From DesktopIcons.tsx:5-42
const useDeviceType = () => {
  const [deviceType, setDeviceType] = useState<'desktop' | 'tablet' | 'mobile'>('desktop');
  
  useEffect(() => {
    const checkDeviceType = () => {
      const width = window.innerWidth;
      
      if (width < 640) {
        setDeviceType('mobile');
      } else if (width < 1024) {
        setDeviceType('tablet');
      } else {
        setDeviceType('desktop');
      }
    };
    
    checkDeviceType();
    window.addEventListener('resize', checkDeviceType);
    return () => window.removeEventListener('resize', checkDeviceType);
  }, []);
  
  return { deviceType, windowSize };
};
```

## Code Example: Main Page Integration

```tsx
// From app/page.tsx:19-86
export default function Home() {
  const [isCLI, setIsCLI] = useState(false);
  const [wallpaper, setWallpaper] = useState<string | null>(null);
  
  const handleWallpaperChange = (newWallpaper: string) => {
    setWallpaper(newWallpaper);
  };
  
  const backgroundStyle = wallpaper
    ? {
        backgroundImage: `url(${wallpaper})`,
        backgroundSize: 'cover',
        backgroundPosition: 'center',
        backgroundRepeat: 'no-repeat',
      }
    : {};
  
  return (
    <div className="relative h-screen w-full" style={backgroundStyle}>
      {/* Quote component (desktop mode only) */}
      {!isCLI && <Quote />}
      
      {/* Desktop icons */}
      <div className="relative h-full flex flex-col">
        {!isCLI && <DesktopIcons onWallpaperChange={handleWallpaperChange} />}
        
        {/* Toggle button */}
        <div className="flex-grow flex items-center justify-center">
          <ToggleButton isCLI={isCLI} onToggle={() => setIsCLI(!isCLI)} />
        </div>
        
        {/* Floating dock (desktop mode only) */}
        {!isCLI && (
          <div className="fixed bottom-4 sm:bottom-8 left-0 right-0 flex justify-center z-40">
            <FloatingDockDemo onWallpaperChange={handleWallpaperChange} />
          </div>
        )}
      </div>
    </div>
  );
}
```
