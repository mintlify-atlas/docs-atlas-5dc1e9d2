## Overview

The terminal component (`Terminal.tsx`) is highly customizable. You can add new commands, modify the file system structure, change styling, and extend functionality to match your portfolio needs.

## Adding New Commands

### Basic Command Structure

Commands are handled in the `processCommand` function within `Terminal.tsx`. Add new cases to the switch statement:

```typescript Terminal.tsx:184-441
switch (command) {
  case 'your-command':
    // Command logic here
    output = 'Your output';
    break;
  // ... other cases
}
```

### Command with Arguments

Create a command that accepts and processes arguments:

```typescript Terminal.tsx
case 'greet':
  if (args.length === 0) {
    output = 'Usage: greet [name]';
  } else {
    const name = args.join(' ');
    output = `Hello, ${name}! Welcome to my portfolio!`;
  }
  break;
```

**Usage**:
```bash
guest@portfolio:~$ greet
Usage: greet [name]

guest@portfolio:~$ greet John Doe
Hello, John Doe! Welcome to my portfolio!
```

### Interactive Command with API Call

Create a command that fetches data from an API:

```typescript Terminal.tsx
case 'weather':
  if (args.length === 0) {
    output = 'Usage: weather [city]';
  } else {
    const city = args.join(' ');
    output = 'Fetching weather...';
    
    // Async operation
    (async () => {
      try {
        const response = await fetch(
          `https://api.openweathermap.org/data/2.5/weather?q=${city}&appid=YOUR_API_KEY`
        );
        const data = await response.json();
        
        // Update output with weather data
        setCommands(prev => {
          const updated = [...prev];
          updated[updated.length - 1].output = (
            <div>
              <div>Weather in {data.name}:</div>
              <div>Temperature: {(data.main.temp - 273.15).toFixed(1)}°C</div>
              <div>Conditions: {data.weather[0].description}</div>
            </div>
          );
          return updated;
        });
      } catch (error) {
        setCommands(prev => {
          const updated = [...prev];
          updated[updated.length - 1].output = 'Error fetching weather data';
          return updated;
        });
      }
    })();
  }
  break;
```

## Advanced Customization

### Auto-Completion Enhancement

Improve the tab completion functionality:

```typescript Terminal.tsx:133-146
const autoComplete = () => {
  const parts = input.trim().split(' ');
  const command = parts[0];
  const lastPart = parts[parts.length - 1];
  const currentPath = getCurrentPath();

  // Complete command names
  if (parts.length === 1) {
    const commands = [
      'ls', 'cd', 'pwd', 'cat', 'mkdir', 'touch', 'rm',
      'whoami', 'date', 'uname', 'history', 'clear',
      'echo', 'experience', 'github', 'help'
    ];
    
    const matches = commands.filter(cmd => cmd.startsWith(lastPart));
    if (matches.length === 1) {
      setInput(matches[0]);
    }
    return;
  }

  // Complete file/directory names
  if (currentPath.contents) {
    const matches = Object.keys(currentPath.contents)
      .filter(name => name.startsWith(lastPart));

    if (matches.length === 1) {
      parts[parts.length - 1] = matches[0];
      setInput(parts.join(' '));
    } else if (matches.length > 1) {
      // Show all matches
      setCommands(prev => [
        ...prev,
        {
          id: Date.now(),
          input: input,
          time: getFormattedTime(),
          output: matches.join('  '),
          directory: currentDirectory,
        }
      ]);
    }
  }
};
```

### Custom Command Parser

Implement advanced command parsing:

```typescript Terminal.tsx
const parseCommand = (input: string) => {
  // Handle quoted arguments
  const regex = /[^\s"]+|"([^"]*)"/g;
  const parts = [];
  let match;
  
  while ((match = regex.exec(input)) !== null) {
    parts.push(match[1] ? match[1] : match[0]);
  }
  
  return {
    command: parts[0]?.toLowerCase() || '',
    args: parts.slice(1),
    flags: parts.filter(p => p.startsWith('-'))
  };
};

// Use in processCommand
const processCommand = async (e: KeyboardEvent<HTMLInputElement>) => {
  e.preventDefault();
  const { command, args, flags } = parseCommand(input.trim());
  
  // Now you can check for flags
  switch (command) {
    case 'ls':
      const showHidden = flags.includes('-a');
      const longFormat = flags.includes('-l');
      // ...
      break;
  }
};
```

## Configuration File Approach

Create a separate configuration file for easy management:

```typescript config/terminal-config.ts
export const terminalConfig = {
  // File system structure
  fileSystem: {
    '~': {
      type: 'directory',
      contents: {
        'about.txt': { type: 'file', content: '...' },
        // ... more files
      }
    }
  },
  
  // Available commands
  commands: [
    'ls', 'cd', 'pwd', 'cat', 'mkdir', 'touch', 'rm',
    'whoami', 'date', 'uname', 'history', 'clear',
    'echo', 'experience', 'github', 'help'
  ],
  
  // Styling
  theme: {
    backgroundColor: 'bg-black/90',
    borderColor: 'border-neutral-800',
    textColor: 'text-green-100',
    promptColor: 'text-green-500',
    directoryColor: 'text-blue-400',
  },
  
  // Prompt format
  promptFormat: (user: string, dir: string) => 
    `${user}@portfolio:${dir}$`,
};
```

Then import and use in `Terminal.tsx`:

```typescript Terminal.tsx
import { terminalConfig } from '@/config/terminal-config';

const [fileSystem, setFileSystem] = useState(terminalConfig.fileSystem);
```
