The portfolio includes multiple window components that provide different views and functionality. All windows are built using the `WindowWrapper` component for consistent behavior.

## WindowWrapper Component

The base component that provides window functionality for all window types.

### Props

<ParamField path="children" type="React.ReactNode" required>
  The content to render inside the window.
</ParamField>

<ParamField path="isMaximized" type="boolean" required>
  Whether the window is in maximized state.
</ParamField>

<ParamField path="onClose" type="() => void" required>
  Callback function when window is closed.
</ParamField>

<ParamField path="onMinimize" type="() => void" required>
  Callback function when window is minimized.
</ParamField>

<ParamField path="initialWidth" type="number" default="800">
  Initial width of the window in pixels.
</ParamField>

<ParamField path="initialHeight" type="number" default="600">
  Initial height of the window in pixels.
</ParamField>

### Example Usage

```typescript
import { WindowWrapper } from '@/components/ui/WindowWrapper';
import { useState } from 'react';

function MyWindow({ onClose }: { onClose: () => void }) {
  const [isMaximized, setIsMaximized] = useState(false);
  const [isMinimized, setIsMinimized] = useState(false);

  return (
    <WindowWrapper
      isMaximized={isMaximized}
      onClose={onClose}
      onMinimize={() => setIsMinimized(true)}
      initialWidth={800}
      initialHeight={600}
    >
      <div className="h-full flex flex-col">
        {/* Window header */}
        <div className="h-12 bg-gradient-to-r from-gray-800/50 to-gray-900/50">
          {/* Header content */}
        </div>
        
        {/* Window content */}
        <div className="flex-1 overflow-auto">
          {/* Your content here */}
        </div>
      </div>
    </WindowWrapper>
  );
}
```

### Features

- **Draggable**: Drag window by header (disabled when maximized)
- **Resizable**: Resize from all edges and corners (disabled when maximized)
- **Mobile Responsive**: Automatically fullscreen on mobile devices
- **Smooth Animations**: Framer Motion animations for open/close/maximize
- **Constrained Movement**: Windows stay within viewport boundaries

## Window Header Pattern

All windows follow a consistent header pattern:

```typescript
<motion.div
  className="h-12 bg-gradient-to-r from-gray-800/50 to-gray-900/50 flex items-center justify-between px-4 cursor-move border-b border-white/10"
  onPointerDown={e => !isMaximized && dragControls.start(e)}
>
  <div className="flex items-center gap-3">
    <div className="bg-blue-500/20 p-1.5 rounded-full">
      <IconUser size={16} className="text-blue-400" />
    </div>
    <span className="text-white/90 text-sm font-medium">Window Title</span>
  </div>
  
  {/* Window Controls */}
  <div className="flex items-center gap-1">
    <motion.button
      whileHover={{ backgroundColor: 'rgba(107, 114, 128, 0.2)' }}
      className="p-2 rounded-full"
    >
      <IconMinus size={14} className="text-white/80" />
    </motion.button>
    <motion.button
      whileHover={{ backgroundColor: 'rgba(107, 114, 128, 0.2)' }}
      onClick={() => setIsMaximized(!isMaximized)}
      className="p-2 rounded-full"
    >
      <IconSquare size={14} className="text-white/80" />
    </motion.button>
    <motion.button
      whileHover={{ backgroundColor: 'rgba(239, 68, 68, 0.2)' }}
      onClick={onClose}
      className="p-2 rounded-full"
    >
      <IconX size={14} className="text-white/80" />
    </motion.button>
  </div>
</motion.div>
```

## Responsive Behavior

Windows automatically adapt to mobile devices:

```typescript
const useIsMobile = () => {
  const [isMobile, setIsMobile] = useState(false);

  useEffect(() => {
    const checkIsMobile = () => {
      setIsMobile(window.innerWidth < 768);
    };

    checkIsMobile();
    window.addEventListener('resize', checkIsMobile);
    return () => window.removeEventListener('resize', checkIsMobile);
  }, []);

  return isMobile;
};

// Force fullscreen on mobile
const shouldBeFullscreen = isMobile || isMaximized;
```

## Animation Configuration

```typescript
<motion.div
  initial={{ opacity: 0, scale: isMobile ? 1 : 0.98 }}
  animate={{
    opacity: 1,
    scale: 1,
    x: shouldBeFullscreen ? 0 : position.x,
    y: shouldBeFullscreen ? 0 : position.y,
    width: shouldBeFullscreen ? '100%' : size.width,
    height: shouldBeFullscreen ? '100%' : size.height,
  }}
  transition={{
    type: 'spring',
    stiffness: 300,
    damping: 25,
    mass: 0.5,
  }}
/>
```

## Resize Configuration

```typescript
<Resizable
  size={size}
  onResizeStop={(e, direction, ref, d) => {
    setSize({
      width: size.width + d.width,
      height: size.height + d.height,
    });
  }}
  enable={{
    top: true,
    right: true,
    bottom: true,
    left: true,
    topRight: true,
    bottomRight: true,
    bottomLeft: true,
    topLeft: true,
  }}
  minWidth={400}
  minHeight={300}
>
  {children}
</Resizable>
```
